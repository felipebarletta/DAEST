## Media = somatório x_i/n - Função para méria aritmética
media <- function(x){
    soma = 0
    for(i in 1:length(x))
        soma = soma + x[i]
    media = soma/length(x)
    return(media)
}

Intervalo de confiança

# LI = media - Z*desvio_padrão/raiz(n)
# LS = media + Z*desvio_padrão/raiz(n)
IC <- function(x, conf = 0.95){
    avg = media(x)
    z = c(-qnorm(conf),qnorm(conf))
    dp = sd(x)
    n = length(x)
    raizn <- sqrt(n)
    #li = avg - z*(dp/raizn)
    #ls = avg + z*(dp/raizn)
    ic  <- avg + z*(dp/raizn)
    return(paste("IC(95%)= (",  round(ic[1],2),";",round(ic[2],2),")"))
}
IC(x)
[1] "IC(95%)= ( -0.08 ; 0.26 )"

União e intersecção

## Espaço amostral= (1,2,3,4,5,6)
A <- 1:4
B <- 1:3
C <- c(2,4,6)
D <- c(2,3,5)

## Função União
uniao <- function(a,b){
    u = a
    for(i in 1:length(b))
        if(!b[i] %in% u){
            u = c(u,b[i])
        }
        return(u)
    }

uniao(A,B)
[1] 1 2 3 4
uniao(A,C)
[1] 1 2 3 4 6
# Função pronta do R
union(A,C)
[1] 1 2 3 4 6
## Função para intersecção

Inter <- function(a,b){
    inte <- NULL
    for(i in 1:length(a))
        if(a[i] %in% b){
            inte  <- c(inte, a[i])
        }
    return(inte)
    }
Inter(a=A, b=B)
[1] 1 2 3
Inter(A,D)
[1] 2 3
# Função pronta do R
intersect(A,D)
[1] 2 3
